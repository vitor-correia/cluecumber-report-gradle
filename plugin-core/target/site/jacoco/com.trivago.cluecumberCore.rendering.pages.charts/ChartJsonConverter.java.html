package com.trivago.cluecumberCore.rendering.pages.charts;

import com.google.gson.Gson;
import com.google.gson.GsonBuilder;
import com.trivago.cluecumberCore.rendering.pages.charts.pojos.Chart;

import javax.inject.Inject;
import javax.inject.Singleton;

@Singleton
public class ChartJsonConverter {

    private final Gson gson;

    @Inject
    public ChartJsonConverter() {
        gson = new GsonBuilder().setPrettyPrinting().create();
    }

    public String convertChartToJson(final Chart chart) {
        return gson.toJson(chart);
    }
}
